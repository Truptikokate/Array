9. Unique Color Shirt

    function Unique_Shirts (arr,N) {
    count=0;
    for(i=0;i< N;i++){
        for(j=0;j< N;j++){
            if(arr[i]==arr[j] && i!=j){
                break;
            }
        }
        if(j===N)
            count++;
        }
    return count;
    }

10. Min and Max

    function arrayMin(arr) {
        min=arr[0];
        for(i=0;i< arr.length;i++){
            if(arr[i]< min){
                min=arr[i];
            }
        }
    return min;
    }
          
    function arrayMax(arr) {
        for(i=0;i< arr.length;i++){
            if(arr[0]< arr[i]){
                arr[0]=arr[i];
            }
        }
    return arr[0];
    }
